int main()

scanf (\\"%d\\", &n);
 
   if (n<=0) /* if n is a negative  number*/

/* Calling function power n times*/
 
   for (i=1; i<=n; i++)

   }
 
printf (\\"The sum of the 1-%d series  of integers raised in %d is %d\\", n,e,  S);

}
 
/*Fuction to calculate a number raised  in power*/
 
int power(int b, int e)
 
{
 
int power=1;
 
int i;
 
if (b<=0)
 
   {

   }
